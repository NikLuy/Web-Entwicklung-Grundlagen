1️⃣ Variablen Deklaration

var, let, const

Drei Möglichkeiten, Variablen zu deklarieren

// VAR - Function-scoped, kann re-deklariert werden (alt) var oldStyle = "Alte Methode"; var oldStyle = "Neu-Deklaration möglich"; // Erlaubt // LET - Block-scoped, kann geändert werden (modern) let userName = "Max"; userName = "Anna"; // Änderung erlaubt // let userName = "Tom"; // Fehler: Neu-Deklaration nicht erlaubt // CONST - Block-scoped, kann nicht geändert werden const PI = 3.14159; // PI = 3.14; // Fehler: Änderung nicht erlaubt const user = { name: "Max" }; user.name = "Anna"; // Erlaubt: Objekt-Eigenschaften änderbar // Scope-Beispiele { let blockScoped = "nur im Block"; const alsoBlockScoped = "auch nur im Block"; } // console.log(blockScoped); // Fehler: nicht verfügbar

2️⃣ Operatoren

Arithmetische Operatoren

let a = 10, b = 3; a + b // 13 - Addition a - b // 7 - Subtraktion a * b // 30 - Multiplikation a / b // 3.33 - Division a % b // 1 - Modulo (Rest) a ** b // 1000 - Potenzierung // Inkrement/Dekrement let x = 5; x++; // 6 (Post-Inkrement) ++x; // 7 (Pre-Inkrement) x--; // 6 (Post-Dekrement) --x; // 5 (Pre-Dekrement)

Vergleichsoperatoren

// Gleichheit 5 == "5" // true - Lose Gleichheit (type coercion) 5 === "5" // false - Strikte Gleichheit (kein type coercion) 5 != "5" // false 5 !== "5" // true // Vergleiche 10 > 5 // true - Größer als 10 < 5 // false - Kleiner als 10 >= 10 // true - Größer oder gleich 5 <= 10 // true - Kleiner oder gleich

Logische Operatoren

let t = true, f = false; t && t // true - AND (beide müssen true sein) t && f // false t || f // true - OR (mindestens eins muss true sein) f || f // false !t // false - NOT (negiert) !f // true // Short-circuit Evaluation let result = null || "default"; // "default" let value = "text" && "andere"; // "andere" // Nullish Coalescing (ES2020) let x = null ?? "default"; // "default" let y = 0 ?? "default"; // 0 (nicht "default"!)

Zuweisungsoperatoren

let x = 10; x += 5; // x = x + 5 (15) x -= 3; // x = x - 3 (12) x *= 2; // x = x * 2 (24) x /= 4; // x = x / 4 (6) x %= 4; // x = x % 4 (2) x **= 3; // x = x ** 3 (8)

3️⃣ If-Else Bedingungen

If, Else If, Else

let age = 18; // Einfaches If if (age >= 18) { console.log("Volljährig"); } // If-Else if (age >= 18) { console.log("Volljährig"); } else { console.log("Minderjährig"); } // If-Else If-Else if (age < 13) { console.log("Kind"); } else if (age < 18) { console.log("Jugendlich"); } else if (age < 65) { console.log("Erwachsen"); } else { console.log("Senior"); } // Ternärer Operator (Kurzform) let status = age >= 18 ? "Erwachsen" : "Minderjährig"; // Verschachtelte Bedingungen if (age >= 18) { if (age <= 25) { console.log("Junger Erwachsener"); } }

4️⃣ Switch Statement

Switch-Case

let day = 3; let dayName; switch (day) { case 1: dayName = "Montag"; break; case 2: dayName = "Dienstag"; break; case 3: dayName = "Mittwoch"; break; case 4: dayName = "Donnerstag"; break; case 5: dayName = "Freitag"; break; case 6: case 7: dayName = "Wochenende"; // Mehrere Cases zusammen break; default: dayName = "Ungültiger Tag"; } // Switch mit Strings let fruit = "Apfel"; switch (fruit) { case "Apfel": console.log("🍎 Rot und knackig"); break; case "Banane": console.log("🍌 Gelb und süß"); break; case "Orange": console.log("🍊 Saftig"); break; default: console.log("🍇 Anderes Obst"); }

5️⃣ Schleifen (Loops)

For-Schleife

// Klassische For-Schleife for (let i = 0; i < 5; i++) { console.log("Durchlauf " + i); } // Rückwärts zählen for (let i = 10; i > 0; i--) { console.log(i); } // Schrittweite ändern for (let i = 0; i <= 10; i += 2) { console.log(i); // 0, 2, 4, 6, 8, 10 } // Verschachtelte Schleifen for (let i = 1; i <= 3; i++) { for (let j = 1; j <= 3; j++) { console.log(`i=${i}, j=${j}`); } }

While & Do-While

// While-Schleife - Prüfung vor Ausführung let count = 0; while (count < 5) { console.log("Count: " + count); count++; } // Do-While-Schleife - Prüfung nach Ausführung let num = 0; do { console.log("Num: " + num); num++; } while (num < 5); // Wird mindestens einmal ausgeführt, auch wenn Bedingung false ist let x = 10; do { console.log("Einmal ausgeführt"); } while (x < 5); // Bedingung ist false, aber Block wird einmal ausgeführt

For...of & For...in

// For...of - Iteriert über Werte (Arrays, Strings, etc.) const fruits = ["Apfel", "Banane", "Orange"]; for (const fruit of fruits) { console.log(fruit); // "Apfel", "Banane", "Orange" } // String iterieren for (const char of "Hello") { console.log(char); // H, e, l, l, o } // For...in - Iteriert über Keys (Objekte, Arrays) const person = { name: "Max", age: 30, city: "Berlin" }; for (const key in person) { console.log(`${key}: ${person[key]}`); } // Array mit for...in (zeigt Indizes) for (const index in fruits) { console.log(`Index ${index}: ${fruits[index]}`); }

Break & Continue

// Break - Beendet die Schleife for (let i = 0; i < 10; i++) { if (i === 5) { break; // Stoppt bei 5 } console.log(i); // 0, 1, 2, 3, 4 } // Continue - Überspringt aktuelle Iteration for (let i = 0; i < 10; i++) { if (i % 2 === 0) { continue; // Überspringt gerade Zahlen } console.log(i); // 1, 3, 5, 7, 9 } // Label für verschachtelte Schleifen outerLoop: for (let i = 0; i < 3; i++) { for (let j = 0; j < 3; j++) { if (i === 1 && j === 1) { break outerLoop; // Beendet äußere Schleife } console.log(`i=${i}, j=${j}`); } }

6️⃣ Funktionen

Funktions-Deklarationen

// Function Declaration function greet(name) { return `Hallo ${name}!`; } // Function Expression const greet2 = function(name) { return `Hi ${name}!`; }; // Arrow Function (ES6) const greet3 = (name) => `Hey ${name}!`; const greet4 = name => `Servus ${name}!`; // Ein Parameter ohne () // Default Parameters function greetWithDefault(name = "Gast") { return `Hallo ${name}!`; } // Rest Parameters function sum(...numbers) { return numbers.reduce((a, b) => a + b, 0); } sum(1, 2, 3, 4, 5); // 15 // Destrukturierung in Parametern function printUser({ name, age }) { console.log(`${name} ist ${age} Jahre alt`); }

Callback-Funktionen & Higher-Order Functions

// Callback-Funktion function processArray(arr, callback) { const result = []; for (let item of arr) { result.push(callback(item)); } return result; } const numbers = [1, 2, 3, 4]; const doubled = processArray(numbers, x => x * 2); // Higher-Order Function (gibt Funktion zurück) function multiplier(factor) { return function(number) { return number * factor; }; } const double = multiplier(2); const triple = multiplier(3); double(5); // 10 triple(5); // 15

7️⃣ Array-Methoden

Wichtige Array-Methoden

const numbers = [1, 2, 3, 4, 5]; // forEach - Iteriert über alle Elemente numbers.forEach((num, index) => { console.log(`Index ${index}: ${num}`); }); // map - Transformiert jedes Element const doubled = numbers.map(n => n * 2); // [2, 4, 6, 8, 10] // filter - Filtert Elemente const evenNumbers = numbers.filter(n => n % 2 === 0); // [2, 4] // reduce - Reduziert zu einem Wert const sum = numbers.reduce((acc, n) => acc + n, 0); // 15 // find - Findet erstes Element const found = numbers.find(n => n > 3); // 4 // some - Prüft ob mindestens eins die Bedingung erfüllt const hasEven = numbers.some(n => n % 2 === 0); // true // every - Prüft ob alle die Bedingung erfüllen const allPositive = numbers.every(n => n > 0); // true // includes - Prüft ob Element enthalten ist numbers.includes(3); // true

8️⃣ String-Methoden

Wichtige String-Operationen

const text = "JavaScript Programmierung"; // Länge und Zugriff text.length; // 24 text[0]; // "J" text.charAt(4); // "S" // Suchen text.indexOf("Script"); // 4 text.lastIndexOf("a"); // Letzte Position text.includes("Java"); // true text.startsWith("Java"); // true text.endsWith("ung"); // true // Extrahieren text.substring(0, 10); // "JavaScript" text.slice(0, 10); // "JavaScript" text.slice(-13); // "Programmierung" // Transformieren text.toLowerCase(); // "javascript programmierung" text.toUpperCase(); // "JAVASCRIPT PROGRAMMIERUNG" " text ".trim(); // "text" text.replace("Java", "Type"); // "TypeScript..." text.replaceAll("a", "@"); // Alle "a" ersetzen // Teilen und Verbinden text.split(" "); // ["JavaScript", "Programmierung"] ["Java", "Script"].join(""); // "JavaScript" // Wiederholen und Auffüllen "Ha".repeat(3); // "HaHaHa" "5".padStart(3, "0"); // "005" "5".padEnd(3, "0"); // "500"

Objekt-Operationen

🔟 Destructuring (ES6)

Array & Object Destructuring

// Array Destructuring const colors = ["Rot", "Grün", "Blau"]; const [first, second, third] = colors; console.log(first); // "Rot" // Rest Operator const [head, ...tail] = [1, 2, 3, 4]; console.log(head); // 1 console.log(tail); // [2, 3, 4] // Object Destructuring const user = { name: "Max", age: 30, city: "Berlin" }; const { name, age } = user; console.log(name); // "Max" // Umbenennen const { name: userName, age: userAge } = user; // Default-Werte const { name, country = "Deutschland" } = user; // Verschachteltes Destructuring const data = { user: { name: "Max", email: "max@test.de" } }; const { user: { email } } = data;

1️⃣1️⃣ Spread & Rest Operator

... Operator

// Spread bei Arrays const arr1 = [1, 2, 3]; const arr2 = [4, 5, 6]; const combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6] // Array kopieren const copy = [...arr1]; // Spread bei Objekten const obj1 = { a: 1, b: 2 }; const obj2 = { c: 3, d: 4 }; const merged = { ...obj1, ...obj2 }; // {a:1, b:2, c:3, d:4} // Objekt erweitern/überschreiben const user = { name: "Max", age: 30 }; const updated = { ...user, age: 31, city: "Berlin" }; // Rest bei Funktionsparametern function sum(...numbers) { return numbers.reduce((a, b) => a + b, 0); } sum(1, 2, 3, 4, 5); // 15 // Rest bei Destructuring const { a, ...rest } = { a: 1, b: 2, c: 3 }; console.log(rest); // { b: 2, c: 3 }

1️⃣4️⃣ This Keyword

Kontext von "this"

// In Methode - bezieht sich auf Objekt const person = { name: "Max", greet() { console.log(`Hallo, ich bin ${this.name}`); } }; person.greet(); // "Hallo, ich bin Max" // Arrow Functions - this vom umgebenden Scope const obj = { name: "Test", regular: function() { console.log(this.name); // "Test" }, arrow: () => { console.log(this.name); // undefined (globales this) } }; // Explicit Binding function greet() { console.log(`Hallo ${this.name}`); } const user = { name: "Anna" }; greet.call(user); // "Hallo Anna" greet.apply(user); // "Hallo Anna" const boundGreet = greet.bind(user); boundGreet(); // "Hallo Anna"
